28  import org.apache.maven.plugin.AbstractMojo;
29  import org.apache.maven.plugin.MojoExecutionException;
30  import org.apache.maven.project.MavenProject;
31  import org.codehaus.plexus.util.cli.CommandLineUtils;

33  /**
34   * The read-project-properties goal reads property files and stores
35   * the properties as project properties.  It serves as an alternate
36   * to specifying properties in pom.xml.
37   *
38   * @author <a href="mailto:zarars@gmail.com">Zarar Siddiqi</a>
39   * @author <a href="mailto:Krystian.Nowak@gmail.com">Krystian Nowak</a>
40   * @version $Id$
41   * @goal read-project-properties
42   */
43  public class ReadPropertiesMojo extends AbstractMojo
44  {
45      /**
46       * @parameter default-value="${project}"
47       * @required
48       * @readonly
49       */
50      private MavenProject project;
51  
52      /**
53       * The properties files that will be used when reading properties.
54       *
55       * @parameter
56       * @required
57       */
58      private File[] files;
59  
60      /**
61       * If the plugin should be quiet if any of the files was not found
62       *
63       * @parameter default-value="false"
64       */
65      private boolean quiet;
66  
67  
68      public void execute()
69          throws MojoExecutionException
70      {
71          Properties projectProperties = new Properties();
72          for ( int i = 0; i < files.length; i++ )
73          {
74              File file = files[i];
75  
76              if ( file.exists() )
77              {
78                  try
79                  {
80                      getLog().debug( "Loading property file: " + file );
81                      
82                      FileInputStream stream = new FileInputStream( file );
83                      projectProperties = project.getProperties();
84                      
85                      try
86                      {
87                          projectProperties.load( stream );
88                      }
89                      finally
90                      {
91                          if ( stream != null )
92                          {
93                              stream.close();
94                          }
95                      }
96                  }
97                  catch ( IOException e )
98                  {
99                      throw new MojoExecutionException( "Error reading properties file " + file.getAbsolutePath(), e );
100                 }
101             }
102             else
103             {
104                 if ( quiet )
105                 {
106                     getLog().warn( "Ignoring missing properties file: " + file.getAbsolutePath() );
107                 }
108                 else
109                 {
110                     throw new MojoExecutionException( "Properties file not found: " + file.getAbsolutePath() );
111                 }
112             }
113         }
114 
115         boolean useEnvVariables = false;
116         for ( Enumeration n = projectProperties.propertyNames(); n.hasMoreElements(); )
117         {
118             String k = (String) n.nextElement();
119             String p = (String) projectProperties.get( k );
120             if ( p.indexOf( "${env." ) != -1 )
121             {
122                 useEnvVariables = true;
123                 break;
124             }
125         }
126         Properties environment = null;
127         if ( useEnvVariables )
128         {
129             try
130             {
131                 environment = CommandLineUtils.getSystemEnvVars();
132             }
133             catch ( IOException e )
134             {
135                 throw new MojoExecutionException( "Error getting system envorinment variables: ", e );
136             }
137         }
138         for ( Enumeration n = projectProperties.propertyNames(); n.hasMoreElements(); )
139         {
140             String k = (String) n.nextElement();
141             projectProperties.setProperty( k, getPropertyValue( k, projectProperties, environment ) );
142         }
143     }

145     /**
146      * Retrieves a property value, replacing values like ${token}
147      * using the Properties to look them up.
148      * Shamelessly adapted from:
149      * http://maven.apache.org/plugins/maven-war-plugin/xref/org/apache/maven/plugin/war/PropertyUtils.html
150      *
151      * It will leave unresolved properties alone, trying for System
152      * properties, and environment variables and implements reparsing
153      * (in the case that the value of a property contains a key), and will
154      * not loop endlessly on a pair like test = ${test}
155      *
156      * @param k           property key
157      * @param p           project properties
158      * @param environment environment variables
159      * @return resolved property value
160      */
161     private String getPropertyValue( String k, Properties p, Properties environment )
162     {
163         String v = p.getProperty( k );
164         String ret = "";
165         int idx, idx2;
166 
167         while ( ( idx = v.indexOf( "${" ) ) >= 0 )
168         {
169             // append prefix to result
170             ret += v.substring( 0, idx );
171 
172             // strip prefix from original
173             v = v.substring( idx + 2 );
174 
175             idx2 = v.indexOf( "}" );
176             
177             // if no matching } then bail
178             if ( idx2 < 0 )
179             {
180                 break;
181             }
182 
183             // strip out the key and resolve it
184             // resolve the key/value for the ${statement}
185             String nk = v.substring( 0, idx2 );
186             v = v.substring( idx2 + 1 );
187             String nv = p.getProperty( nk );
188 
189             // try global environment
190             if ( nv == null )
191             {
192                 nv = System.getProperty( nk );
193             }
194 
195             // try environment variable
196             if ( nv == null && nk.startsWith( "env." ) && environment != null )
197             {
198                 nv = environment.getProperty( nk.substring( 4 ) );
199             }
200 
201             // if the key cannot be resolved,
202             // leave it alone ( and don't parse again )
203             // else prefix the original string with the
204             // resolved property ( so it can be parsed further )
205             // taking recursion into account.
206             if ( nv == null || nv.equals( nk ) )
207             {
208                 ret += "${" + nk + "}";
209             }
210             else
211             {
212                 v = nv + v;
213             }
214         }
215         return ret + v;
216     }
